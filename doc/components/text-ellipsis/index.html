# TextEllipsis 文本省略

按宽度自动显示省略号与Tooltip的文本容器

# 基本用法

在需要动态改变单元格内容的表格列中，无法利用表格组件自带的单行省略与tooltip功能。使用该组件可以实现类似的效果

通过content设置文本内容，通过tooltip设置提示内容

<template>
  <div>
    <el-button @click="switchEdit">开启/关闭编辑</el-button>
    <el-table
      :data="tableData"
      border
    >
      <el-table-column
        prop="tel"
        label="使用span"
      >
        <template slot-scope="scope">
          <div style="display: flex; width: 100%">
            <div v-show="editing">
              <el-input v-model="scope.row.tel"></el-input>
            </div>
            <span v-show="!editing">
              {{ scope.row.tel }}
            </span>
          </div>
        </template>
      </el-table-column>
      <el-table-column
        prop="tel"
        label="使用TextEllipsis"
      >
        <template slot-scope="scope">
          <div style="display: flex; width: 100%">
            <div v-show="editing">
              <el-input v-model="scope.row.tel"></el-input>
            </div>
            <div v-show="!editing" style="flex-grow: 1; width: 0">
              <TextEllipsis :content="scope.row.tel" :tooltip="scope.row.tel" />
            </div>
          </div>
        </template>
      </el-table-column>
      <el-table-column
        prop="tel"
        label="el-table自带的效果"
        show-overflow-tooltip
      >
      </el-table-column>
    </el-table>
  </div>
</template>
<script>
  
export default {
  data() {
    return {
      tableData: [
        {
          tel: '1234567890098765432112345678900987654321'
        },
        {
          tel: '1234567890098765432112345678900987654321'
        }
      ],
      editing: false
    }
  },
  methods: {
    switchEdit() {
        this.editing = !this.editing
    }
  }
}
</script>
