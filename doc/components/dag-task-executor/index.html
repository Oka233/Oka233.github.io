# DAGTaskExecutor 有向图任务执行流水线

有向图任务执行流水线，用于流程图中带有数据依赖的异步任务执行场景

# 基本用法

为图节点定义如下形式的任务，将图传入执行器，使用执行器执行任务

const job = (data, resolve, reject) => {
  /***
   * @param {Array.<Object>} data: 来自前序节点的数据
   * @param {function(Object)} resolve: 任务执行成功回调
   * @param {function(Object)} reject: 任务执行失败回调
   */
  try {
    // 结果中除处理完成的数据外，还应包含来源节点或任务的信息，供后续节点判断。以下是一个示例
    // 该返回结构不是必须的，如果后续节点不需要了解所获取数据的来源，可以只返回数据
    resolve({
      data: process(data),
      source: {
        id: nodeId,
        type: taskType
      }
    })
  } catch (e) {
    reject(e)
  }
}

创建执行器实例，注册任务，绑定事件，执行任务

import DAGTaskExecutor from './DAGTaskExecutor'

const executor = new DAGTaskExecutor()
executor.registerJob('jobName', jobFunction)
executor.bind('eventName', eventHandler)

executor.setGraph(graphInstance)
executor.run()

# Playground

  • 本示例为开放性示例，没有检验无效输入
  • 为演示错误处理与重试，任务10%概率执行失败
  • 在任务未执行完时重置或清空可能导致异常。实际中可通过registerJob保存当前处理的图/任务id，并检查后续事件回调中的id与当前是否一致来避免。这里省略了此步骤

# Playground代码

  • 现使用AntV X6的Graph实例，用于查找节点关系与读取节点上保存的任务，可以替换为其它图实例
<template>
  <div>
    <div class="demonstration tool-bar">
      <el-button @click="execute">执行</el-button>
      <div>
        单步：
        <el-switch v-model="byStep" />
      </div>
      <div>
        并发：
        <el-input-number v-model="limit" :min="1" :max="100" style="width: 140px" />
      </div>
      <el-button @click="reset">重置</el-button>
      <el-button @click="empty">清空</el-button>
    </div>
    <BlueprintEditor ref="blueprint" />
  </div>
</template>

<script>
import BlueprintEditor from './BlueprintEditor'
import DAGTaskExecutor from './DAGTaskExecutor/DAGTaskExecutor'
import demoData from './DAGTaskExecutor/demo'

export default {
  name: 'DAGTEDemo',
  components: {
    BlueprintEditor
  },
  data() {
    return {
      executor: null,
      byStep: false,
      limit: 3,
      running: false
    }
  },
  watch: {
    limit(val) {
      this.executor.setLimit(val)
    }
  },
  mounted() {
    // 初始化执行器，注册任务，绑定事件
    this.executor = new DAGTaskExecutor()
    this.executor.registerJob('demoJob', (data, resolve, reject) => {
      setTimeout(() => {
        if (Math.random() > 0.1) {
          resolve(Date.now())
        } else {
          reject(new Error('运气不好'))
        }
      }, Math.floor(Math.random() * (1000 - 100 + 1) + 100))
    })
    this.executor.bind('task-done', ({ node, children }) => {
      this.$refs.blueprint.highlight(node, 'success')
      if (!children.length) {
        this.$message.success('计算执行成功')
        this.running = false
      }
    })
    this.executor.bind('task-fail', ({ node }, e) => {
      this.$refs.blueprint.highlight(node, 'error')
      this.$message.error(`${node.getAttrs().text.text}执行失败，因${e.message}，可尝试重试执行`)
    })
    this.executor.bind('task-begin', ({ node }) => {
      this.$refs.blueprint.highlight(node, 'running')
    })
    // 加载示例计算图与任务
    this.$refs.blueprint.fromJSON(demoData)
    this.$refs.blueprint.getGraph().on('node:added', ({ node }) => {
      node.setData({
        task: 'demoJob'
      })
    })
  },
  methods: {
    execute() {
      if (!this.running) {
        this.executor.reset()
        this.$refs.blueprint.unhighlightAll()
      }
      // 实际中应视情况setGraph，这里是为了演示方便
      this.executor.setGraph(this.$refs.blueprint.getGraph())
      this.executor.run(this.byStep)
      this.running = true
    },
    reset() {
      this.executor.reset()
      this.$refs.blueprint.fromJSON(demoData)
    },
    empty() {
      this.executor.reset()
      this.$refs.blueprint.fromJSON({ cells: [] })
    }
  }
}
</script>
