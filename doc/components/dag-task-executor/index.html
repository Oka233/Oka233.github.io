# DAGTaskExecutor 有向图任务执行流水线

有向图任务执行流水线

# 基本用法

为图节点定义如下形式的任务，将图传入执行器，使用执行器执行任务

const job = (data, resolve, reject) => {
  /***
   * @param data: 来自前序节点的数据
   * @param resolve: 任务执行成功回调
   * @param reject: 任务执行失败回调
   */
  try {
    newData = nodeTask(data)
    resolve(newData)
  } catch (e) {
    reject(e)
  }
}

# Demo代码

  • 现使用AntV X6的Graph实例，用于查找节点关系与绑定的任务，可以替换为其它图实例
<template>
  <div>
    <div class="demonstration tool-bar">
      <el-button @click="execute">执行</el-button>
      <div>
        单步：
        <el-switch v-model="byStep" />
      </div>
      <div>
        并发：
        <el-input-number v-model="limit" :min="1" :max="100" style="width: 140px" />
      </div>
      <el-button @click="reset">重置</el-button>
      <el-button @click="empty">清空</el-button>
    </div>
    <BlueprintEditor ref="editor" />
  </div>
</template>

<script>
import BlueprintEditor from "./BlueprintEditor";
import DAGTaskExecutor from "./DAGTaskExecutor/DAGTaskExecutor";
import demoData from "./DAGTaskExecutor/demo";

export default {
  name: "DAGTEDemo",
  components: {
    BlueprintEditor
  },
  data() {
    return {
      executor: null,
      byStep: false,
      limit: 3,
      running: false
    }
  },
  mounted() {
    // 初始化执行器，注册任务，绑定事件
    this.executor = new DAGTaskExecutor();
    this.executor.registerJob("basicTask", (data, resolve, reject) => {
      setTimeout(() => {
        if (Math.random() > 0.1) {
          resolve(data);
        } else {
          reject();
        }
      }, Math.floor(Math.random() * (1000 - 100 + 1) + 100));
    });
    this.executor.bind("task-done", ({ node, children }) => {
      this.$refs.editor.highlight(node, 'success');
      if (!children.length) {
        this.$message.success("计算执行成功");
        this.running = false;
      }
    });
    this.executor.bind("task-fail", ({ node }) => {
      this.$refs.editor.highlight(node, 'error');
      this.$message.error(`节点 ${node.getAttrs().text.text} 执行失败，可尝试重试执行`);
    });
    this.executor.bind("task-begin", ({ node }) => {
      this.$refs.editor.highlight(node, 'running');
    });
    // 加载示例计算图与任务
    this.$refs.editor.fromJSON(demoData);
    this.$refs.editor.getGraph().on('node:added', ({ node }) => {
      node.setData({
        task: "basicTask"
      });
    });
  },
  watch: {
    limit(val) {
      this.executor.setLimit(val);
    }
  },
  methods: {
    execute() {
      if (!this.running) this.reset();
      this.executor.setGraph(this.$refs.editor.getGraph());
      this.executor.run(this.byStep);
      this.running = true;
    },
    reset() {
      this.executor.reset();
      this.$refs.editor.unhighlightAll();
    },
    empty() {
      this.executor.reset();
      this.$refs.editor.fromJSON({ cells: [] });
    }
  }
}
</script>
