# HierarchyTree 树形指标体系编辑器

基于树形el-table的层次指标体系编辑组件，支持扩展功能的配置化引入

# 基本用法

定义列配置与默认新节点，使用组件提供的操作进行增删改，实现树形结构的编辑

构建指标体系

  />

export default {

                  type: '',

                  id: '69ef25de-6f9e-417d-b23f-7cf84c63876a'

      ],
      columnOptions: {
        mainColumn: {
          type: 'text',
          prop: 'topic',

# 自定义操作

# 替换默认操作

可以替换原本用于增删改的默认操作

通过customRowOperation设置自定义操作，可以为叶节点与非叶节点单独设置

<template>
  <HierarchyTree
    v-model="tableData"
    :column-options="columnOptions"
    :customRowOperation="customRowOperation"
  />
</template>
<script>
  
export default {
  data() {
    return {
      tableData: [
        {
          topic: '指标C1',
          type: '',
          comment: '',
          id: '4b37c472-cf99-4be9-ad14-4146fa272ec0',
          children: [
            {
              topic: '指标C11',
              type: '',
              comment: '',
              id: '49039200-87ae-43f8-a4f8-a89af0163c11',
              children: [
                {
                  topic: '指标C111',
                  type: '',
                  comment: '',
                  id: '4dcc9e27-bbfd-4f54-9ea1-c7fc7c8986a4'
                },
                {
                  topic: '指标C112',
                  type: '',
                  comment: '',
                  id: '69ef25de-6f9e-417d-b23f-7cf84c63876a'
                }
              ]
            }
          ]
        }
      ],
      columnOptions: {
        mainColumn: {
          type: 'text',
          prop: 'topic',
          label: '指标',
          width: '250',
        },
        extraColumn: [
          {
            type: 'input',
            prop: 'type',
            label: '类型',
          },
          {
            type: 'input',
            prop: 'comment',
            label: '备注',
          }
        ]
      },
      customRowOperation: {
        leafOp: {
          text: '叶节点ID',
          func: (row) => {
            this.$message.success(`叶节点${row.id}`)
          }
        },
        nonLeafOp: {
          text: '非叶节点ID',
          func: (row) => {
            this.$message.success(`非叶节点${row.id}`)
          }
        }
      }
    }
  }
}
</script>
